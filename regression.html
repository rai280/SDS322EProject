def run_energyload_regression(df, X, regression_name):

    Y = enb12[['Y1', 'Y2']]
    
    reg = LinearRegression()
    reg.fit(X, Y)
    reg.coef_, reg.intercept_
    
    y_pred = reg.predict(X)
    
    # Diagnosis plot
    plt.plot(y_pred, Y, '.r')
    plt.plot([0, 50], [0, 50], 'k')
    plt.xlabel('Predicted Thermal Load Size', size = 20)
    plt.ylabel('True Thermal Load Size', size = 20)
    plt.title(regression_name + ' Diagnosis Plot')
    plt.show()
    
    # Calculate the absolute errors
    errors = abs(y_pred - Y)
    
    # Calculate mean absolute percentage error (MAPE)
    mape = 100 * (errors / Y)
    # Calculate and display accuracy
    accuracy = 100 - np.mean(mape)
    print(regression_name + ' Accuracy:', round(accuracy, 2), '%.')
    
    # Calculate mean squared error
    mse = np.mean((Y - y_pred)**2)
    print(regression_name + ' MSE:')
    print(mse)
    
    return reg
